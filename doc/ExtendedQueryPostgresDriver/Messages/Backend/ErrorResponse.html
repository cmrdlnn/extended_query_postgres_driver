def initialize(socket)
  super
  message = @content.split(0.chr).each_with_object([]) do |part, memo|
    field = ERROR_FIELDS[part[0]]
    memo << "#{field}: #{part[1..part.size]}" if field
  end
  raise message.join(', ')
end
