def initialize(args)
  @content      = [8, PROTOCOL_VERSION]
  @length_index = 0
  @template     = 'NN'

  ARGUMENTS.each do |arg|
    write_strings(arg.to_s, args[arg]) if args[arg]
  end
  write_options(args[:options]) if args[:options]
  write_byte(0)
end
